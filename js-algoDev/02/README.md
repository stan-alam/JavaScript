```js
if (this.head == this.tail) {
  //only one node in linkedList
  this.head = null;
  this.tail = null;

  return deletedTail;
}
```

```js
//cb21.B
let currentNode = this.head;
while (currentNode.next) {
  if(!currentNode.next.next) {
    currentNode.next = null;
  } else {
    currentNode = currentNode;
  }

this.tail = currentNode;

return deletedTail;
}
```
